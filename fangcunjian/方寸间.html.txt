// ===== State =====

let searchQuery = "";
let sortBy = "newest";
let userName = "阿颖";
let claudeName = "涟言";

// ===== DOM refs =====
const $ = (id) => document.getElementById(id);
const uploadScreen = $("upload-screen");

// ===== Utilities =====
function formatDate(dateStr) {
  if (!dateStr) return "";
  const d = new Date(dateStr);
  if (isNaN(d)) return "";
  return d.toLocaleDateString("zh-CN", { year: "numeric", month: "2-digit", day: "2-digit" });
}

function formatTime(dateStr) {
  if (!dateStr) return "";
  const d = new Date(dateStr);
  if (isNaN(d)) return "";
  return d.toLocaleTimeString("zh-CN", { hour: "2-digit", minute: "2-digit" });
}

function getMessages(conv) {
  // Try different field names that Claude export might use
  return conv.chat_messages || conv.messages || [];
}

function getMessageText(msg) {
  // Direct text field
  if (typeof msg.text === "string" && msg.text.trim()) return msg.text;
  // Content array (Claude format)
  if (msg.content && Array.isArray(msg.content)) {
    const texts = msg.content.filter(c => c.type === "text" && c.text).map(c => c.text);
    if (texts.length) return texts.join("\n");
  }
  // Content as string
  if (typeof msg.content === "string" && msg.content.trim()) return msg.content;
  // Body field
  if (typeof msg.body === "string" && msg.body.trim()) return msg.body;
  return "";
}

function hasVisibleContent(conv) {
  const msgs = getMessages(conv);
  return msgs.some(m => getMessageText(m).trim().length > 0);
}

function countWords(text) {
  if (!text) return 0;
  const chinese = (text.match(/[\u4e00-\u9fff]/g) || []).length;
  const english = text.replace(/[\u4e00-\u9fff]/g, " ").split(/\s+/).filter(w => w.length > 0).length;
  return chinese + english;
}

function getStats(conv) {
  const msgs = getMessages(conv);
  let totalWords = 0, userWords = 0, claudeWords = 0, userMsgs = 0, claudeMsgs = 0;
  msgs.forEach(m => {
    const text = getMessageText(m);
    if (!text.trim()) return; // skip empty messages
    const wc = countWords(text);
    totalWords += wc;
    if (m.sender === "human") { userWords += wc; userMsgs++; }
    else { claudeWords += wc; claudeMsgs++; }
  });
  const visibleCount = userMsgs + claudeMsgs;
  return {
    totalMessages: visibleCount, userMsgs, claudeMsgs,
    totalWords, userWords, claudeWords,
    firstDate: msgs[0]?.created_at || conv.created_at,
    lastDate: msgs[msgs.length - 1]?.created_at || conv.updated_at,
  };
}

function escapeHtml(text) {
  const div = document.createElement("div");
  div.textContent = text;
  return div.innerHTML;
}

function highlightText(text, query) {
  if (!query || !query.trim()) return escapeHtml(text);
  const escaped = escapeHtml(text);
  const q = query.trim().replace(/[.*+?^${}()|[\]\\]/g, "\\$&");
  const regex = new RegExp(`(${q})`, "gi");
  return escaped.replace(regex, "<mark>$1</mark>");
}

function downloadText(content, filename) {
  const blob = new Blob([content], { type: "text/plain;charset=utf-8" });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = filename;
  a.click();
  URL.revokeObjectURL(url);
}

function exportConv(conv) {
  const stats = getStats(conv);
  const msgs = getMessages(conv);
  let txt = `${conv.name || "未命名对话"}\n${"=".repeat(40)}\n`;
  txt += `时间：${formatDate(stats.firstDate)} ~ ${formatDate(stats.lastDate)}\n`;
  txt += `消息数：${stats.totalMessages} | 字数：${stats.totalWords}\n${"=".repeat(40)}\n\n`;
  msgs.forEach(m => {
    const name = m.sender === "human" ? userName : claudeName;
    const text = getMessageText(m);
    txt += `[${name}] ${formatTime(m.created_at)}\n${text}\n\n`;
  });
  return txt;
}

// ===== Render =====
function getFilteredSorted() {
  let list = [...conversations];
  if (searchQuery.trim()) {
    const q = searchQuery.toLowerCase();
    list = list.filter(conv => {
      if ((conv.name || "").toLowerCase().includes(q)) return true;
      return (getMessages(conv)).some(m => getMessageText(m).toLowerCase().includes(q));
    });
  }
  list.sort((a, b) => {
    if (sortBy === "newest") return new Date(b.created_at || 0) - new Date(a.created_at || 0);
    if (sortBy === "oldest") return new Date(a.created_at || 0) - new Date(b.created_at || 0);
    if (sortBy === "most_messages") return (getMessages(b)?.length || 0) - (getMessages(a)?.length || 0);
    return 0;
  });
  return list;
}

function renderList() {
  const list = getFilteredSorted();
  convCount.textContent = `共 ${conversations.length} 个对话` + (searchQuery.trim() ? ` · 筛选出 ${list.length} 个` : "");

  if (list.length === 0) {
    convList.innerHTML = '<div class="conv-list-empty">没有找到匹配的对话</div>';
    return;
  }

  convList.innerHTML = list.map(conv => {
    const stats = getStats(conv);
    const active = conv.uuid === selectedUuid ? " active" : "";
    return `<div class="conv-item${active}" data-uuid="${conv.uuid}">
      <div class="conv-item-name">${escapeHtml(conv.name || "未命名对话")}</div>
      <div class="conv-item-meta">
        <span>${formatDate(stats.firstDate)}</span>
        <span>·</span>
        <span>${stats.totalMessages} 条</span>
        <span>·</span>
        <span>${stats.totalWords} 字</span>
      </div>
    </div>`;
  }).join("");

  // Bind clicks
  convList.querySelectorAll(".conv-item").forEach(el => {
    el.addEventListener("click", () => {
      selectedUuid = el.dataset.uuid;
      renderList();
      renderChat();
      // Mobile: show chat panel
      chatPanel.classList.add("open");
    });
  });
}

function renderChat() {
  const conv = conversations.find(c => c.uuid === selectedUuid);
  if (!conv) {
    chatEmpty.classList.remove("hidden");
    chatHeader.classList.add("hidden");
    chatMessages.classList.add("hidden");
    return;
  }

  chatEmpty.classList.add("hidden");
  chatHeader.classList.remove("hidden");
  chatMessages.classList.remove("hidden");

  const stats = getStats(conv);
  chatTitle.textContent = conv.name || "未命名对话";
  chatMeta.innerHTML = `
    <span>${formatDate(stats.firstDate)} ~ ${formatDate(stats.lastDate)}</span>
    <span>${stats.totalMessages} 条</span>
    <span>${stats.totalWords} 字</span>
    <span>${userName}: ${stats.userWords} 字 / ${claudeName}: ${stats.claudeWords} 字</span>
  `;

  const msgs = getMessages(conv);
  chatMessages.innerHTML = msgs.map(m => {
    const isUser = m.sender === "human";
    const text = getMessageText(m);
    if (!text.trim()) return "";
    const cls = isUser ? "user" : "claude";
    const name = isUser ? userName : claudeName;
    return `<div class="msg ${cls}">
      <div class="msg-meta">
        <span class="name">${escapeHtml(name)}</span>
        <span>${formatTime(m.created_at)}</span>
      </div>
      <div class="msg-bubble">${highlightText(text, searchQuery)}</div>
    </div>`;
  }).join("");

  chatMessages.scrollTop = 0;
}

// ===== File Loading =====
function loadFile(file) {
  if (!file) return;
  const reader = new FileReader();
  reader.onload = (e) => {
    try {
      const data = JSON.parse(e.target.result);
      if (!Array.isArray(data)) {
        alert("文件格式不对，需要是 conversations.json（数组格式）");
        return;
      }
      conversations = data.filter(conv => {
        const msgs = getMessages(conv);
        // Keep only conversations that have at least one message with visible text
        return msgs.length > 0 && msgs.some(m => getMessageText(m).trim().length > 0);
      });
      selectedUuid = null;
      searchQuery = "";
      searchInput.value = "";
      uploadScreen.style.display = "none";
      mainScreen.classList.add("active");
      renderList();
      renderChat();
    } catch (err) {
      alert("JSON 解析失败：" + err.message);
    }
  };
  reader.readAsText(file);
}

// ===== Events =====

// Upload
uploadZone.addEventListener("click", () => fileInput.click());
uploadZone.addEventListener("dragover", (e) => { e.preventDefault(); uploadZone.classList.add("dragging"); });
uploadZone.addEventListener("dragleave", () => uploadZone.classList.remove("dragging"));
uploadZone.addEventListener("drop", (e) => {
  e.preventDefault();
  uploadZone.classList.remove("dragging");
  loadFile(e.dataTransfer.files[0]);
});
fileInput.addEventListener("change", (e) => loadFile(e.target.files[0]));

// Search
let searchTimer;
searchInput.addEventListener("input", (e) => {
  clearTimeout(searchTimer);
  searchTimer = setTimeout(() => {
    searchQuery = e.target.value;
    renderList();
    if (selectedUuid) renderChat();
  }, 200);
});

// Sort
document.querySelectorAll(".sort-btn").forEach(btn => {
  btn.addEventListener("click", () => {
    sortBy = btn.dataset.sort;
    document.querySelectorAll(".sort-btn").forEach(b => b.classList.remove("active"));
    btn.classList.add("active");
    renderList();
  });
});

// Export single
$("export-btn").addEventListener("click", () => {
  const conv = conversations.find(c => c.uuid === selectedUuid);
  if (!conv) return;
  downloadText(exportConv(conv), `${conv.name || "对话"}.txt`);
});

// Export all
$("export-all-btn").addEventListener("click", () => {
  if (!conversations.length) return;
  let all = "";
  conversations.forEach((conv, i) => {
    if (i > 0) all += "\n\n" + "═".repeat(50) + "\n\n";
    all += exportConv(conv);
  });
  downloadText(all, "所有对话.txt");
});

// Close
$("close-btn").addEventListener("click", () => {
  conversations = [];
  selectedUuid = null;
  mainScreen.classList.remove("active");
  uploadScreen.style.display = "flex";
  fileInput.value = "";
});

// Back (mobile)
$("back-btn").addEventListener("click", () => {
  chatPanel.classList.remove("open");
});

// Settings
$("settings-btn").addEventListener("click", () => {
  $("input-user-name").value = userName;
  $("input-claude-name").value = claudeName;
  settingsModal.classList.remove("hidden");
});

$("modal-cancel").addEventListener("click", () => settingsModal.classList.add("hidden"));

settingsModal.addEventListener("click", (e) => {
  if (e.target === settingsModal) settingsModal.classList.add("hidden");
});

$("modal-save").addEventListener("click", () => {
  userName = $("input-user-name").value || "User";
  claudeName = $("input-claude-name").value || "Claude";
  settingsModal.classList.add("hidden");
  if (selectedUuid) renderChat();
});
</script>
